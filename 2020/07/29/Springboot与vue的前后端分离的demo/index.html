<script>
    const app =new Vue({
        el:"#app",
        data:{
            users:[],
            user:{},
            searchName:"",
            searchPhone:"",
        },
        methods:{
            searchLike(){
                let _this=this
                console.log(this.searchName);
                console.log(this.searchPhone);
                axios.get("http://localhost:8080/user/findLike?name="+this.searchName+"&phone="+this.searchPhone).then(function (response) {
                    console.log(response.data);
                    _this.users=response.data
                })
            },
            updateUserInfo(id){
                let _this=this;
                console.log(id);
                axios.get("http://localhost:8080/user/findById?id="+id).then(function (response) {
                    console.log(response.data);
                    _this.user=response.data;
                })
            },
            deleteUserInfo(id){
                let _this=this;
                if (window.confirm("是否要删除")){
                    console.log(id);
                    axios.get("http://localhost:8080/user/delete?id="+id).then(function (response) {
                        console.log(response.data);

                        if (response.data.success){
                            _this.findAll();
                        }else{
                            alert(response.data.success)
                        }
                    }).catch(function (err) {
                        console.log(err)
                    })
                }
                },

            clear(){
              this.user={}
            },

            a(){
                let _this = this;
                console.log(this.user)
                axios.post("http://localhost:8080/user/save",this.user).then(function (response) {
                    console.log(response.data);

                    if (response.data.success){

                        _this.findAll();
                        _this.user={};
                    }else{
                        alert(response.data.message)
                    }

                }).catch(function (err) {
                    console.log(err)
                })
            },
            findAll(){
                let _this = this;
                //发送查询所有的信息
                axios.get("http://localhost:8080/user/findAll").then(function (response) {
                    _this.users=response.data;
                }).catch(function (err) {
                    console.log(err)
                });
            }

        },
        created(){
            let _this = this;
            _this.findAll();
        }

    })
</script>

控制层

package com.zwq.controller;

import com.zwq.entity.User;
import com.zwq.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.util.StringUtils;
import org.springframework.web.bind.annotation.*;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

@RequestMapping("user")
@RestController
@CrossOrigin
public class UserController {

    @Autowired
    private UserService userService;

    //模糊查询，这些用户信息
    @GetMapping("findLike")
    public List<User> findNameOrPhone(String name ,String phone){
        System.out.println(name);
        System.out.println(phone);
        return userService.findNameOrPhone(name,phone);
    }

    //听过Id查找一个用户
    @GetMapping("findById")
    public User findById(String id){
        return userService.findById(id);
    }

    //通过Id删除一个用户
    @GetMapping("delete")
    public Map<String ,Object> delete(String id){
        userService.delete(id);
        Map<String, Object> map=new HashMap<>();
        try{
            map.put("success",true);
        }catch (Exception e){
            map.put("success",false);
            map.put("message","删除失败");
        }
        return map;
    }


    //实现查找所有
    @GetMapping("/findAll")
    public List<User> findAll(){
        List<User> users = userService.findAll();
        for (User user :users){
            System.out.println(user);
        }
        return users;
    }

    //保存用户，如果用户的Id不为空就进行修改操作，否则就进行保存操作
    //map用作是否操作成功
    @RequestMapping("save")
    public Map<String ,Object> save(@RequestBody  User user){
        Map<String, Object> map= new HashMap<>();
        try{
            if (StringUtils.isEmpty(user.getId())){
                userService.save(user);
            }else{
                userService.update(user);
            }
            map.put("success",true);
        }catch (Exception e){
            e.printStackTrace();
            map.put("success",false);
            map.put("message","添加失败");
        }
        System.out.println(map.get("success"));
        return map;
    }
}

map映射文件

用动态sql来进行模糊查询，这里的id我用的时UUId自动生成的，也可以使用
useGeneratedKeys=”true” keyProperty=”id”将id设置为自增的。

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.zwq.dao.UserDao">

    <select id="findNameOrPhone" resultType="User">
        select id,name,age,salary,phone from t_user
        <where>
            <if test="name!=''">
                name like concat(#{name},'%')
            </if>
            <if test="phone!=''">
               or phone like concat('%',#{phone},'%')
            </if>
        </where>
    </select>

    <update id="update" parameterType="com.zwq.entity.User">
        update t_user set name=#{name},age=#{age},salary=#{salary},phone=#{phone} where id=#{id}
    </update>

    <select id="findById" resultType="com.zwq.entity.User">
        select id,name,age,salary,phone from t_user where id=#{id};
    </select>

    <delete id="delete" parameterType="String">
        delete from t_user where id=#{id}
    </delete>

    <select id="findAll" resultType="com.zwq.entity.User">
        select id  , name ,age ,salary ,phone from t_user ;
    </select>

    <insert id="save" parameterType="com.zwq.entity.User">
        insert into t_user  values(#{id},#{name},#{age},#{salary},#{phone});
    </insert>
</mapper>

总结

查询所有操作：再vue的生命周期函数的created()是加载用户信息此时的用户data属性和method方法已经加载了。再用axios接受后端查询出来的数据。赋给responce，再利用v-model的双向绑定进行赋值。
用户的添加将在页面的数据封装成user实例，传到后台，保存到数据库中。拿到后台的数据后再次调用查询所有的函数，刷新页面。
用户的删除拿到用户的Id属性，根据id到后台删除相应的数据。
用户的修改功能：将用户信息展示到表单中，再次提交到数据中，
用户的搜索：利用mysql中的模糊查询功能，mybatis中的动态sql，将前台的数据利用axios传到后端。
